```bash
crontab -l

# 查看 CRON 日志
tail -f ~/.openclaw/workspace/china-stock-team/logs/cron_*.log
```

### 方法2: Python 调度器

```bash
# 后台运行调度器
cd ~/.openclaw/workspace/china-stock-team
source venv/bin/activate
nohup python scripts/scheduler.py start > logs/scheduler.log 2>&1 &

# 查看状态
python scripts/scheduler.py status

# 停止
python scripts/scheduler.py stop
```
